# Overview

Avatars are user's face image that you can use throughout the design from tables to dialog or menus with options for sizes and shapes.

<Canvas>
    <Story
        name="Avatar Controls"
        argTypes={{
            size: {
                control: {
                    type: `select`,
                },
            },
            shape: {
                control: {
                    type: `select`,
                },
            },
            initialText: {
                control: {
                    type: `text`,
                },
            },
            status: {
                control: {
                    type: `text`,
                },
            },
        }}
        args={{
            size: `default`,
            shape: `square`,
            altText: `user avatar`,
        }}
    >
        {(args) => {
            const { imageUrl, ...restArgs } = args;
            const img = imageUrl || image;
            return <Avatar imageUrl={img} {...restArgs} />;
        }}
    </Story>
</Canvas>
